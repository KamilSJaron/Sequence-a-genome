		tail / head
		cat   # catenate
		less  # > more

		wc    # word count
		tr    # translate
		grep  # global regular expression print
	\end{verbatim}

\begin{frame}[fragile]
	\huge
	\verb!wc -c!
	\Large

		  tr -d "\n" | wc -c


		38
	\end{verbatim}
\end{frame}

\begin{frame}[fragile]
	\huge
	Stream redirection overview
	\Large
	\begin{verbatim}


		grep ">" seq.fasta         # print
		grep ">" seq.fasta | wc    # pipe
		grep ">" seq.fasta > file  # write
		grep ">" seq.fasta >> file # append
	\end{verbatim}
\end{frame}
