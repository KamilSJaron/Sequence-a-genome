\begin{frame}[fragile]
\LARGE
\begin{verbatim}
bsub -n 2 -J "QC" \
'fastqc -t 2 -o out/ \
<sample>_R1.fastq.gz \
<sample>_R2.fastq.gz'
# -q ... queue
# -n ... number of cores
# -J ... job name
\end{verbatim}
\end{frame}

\begin{frame}[fragile]

\LARGE
\begin{verbatim}

'fastqc -t 2 -o out/ \
<sample>_R1.fastq.gz \
<sample>_R2.fastq.gz &> qc.log'
\end{verbatim}
\end{frame}
